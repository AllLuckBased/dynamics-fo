Withholding Tax ERP Special Character Check Pilot Validation




Withholding Tax Code----

UPDATE PWC_E_WITHHOLDINGTAXCODE
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'WITHHOLDINGCODE has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_WITHHOLDINGTAXCODE
	WHERE 
	WITHHOLDINGCODE LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_WITHHOLDINGTAXCODE
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'WITHHOLDINGTAXNAME has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_WITHHOLDINGTAXCODE
	WHERE 
	WITHHOLDINGTAXNAME LIKE '%[^A-Za-z0-9 -]%'
