CURRENCIES---
Validation rules--
CURRENCYCODE is blank --- Error --- 2
REFERENCECURRENCYFORTRIANGULATION is blank --- Warning --- 3
GENERALROUNDINGRULE is blank --- Warning --- 3
SYMBOL is blank --- Warning --- 3
NAME is blank --- Warning --- 3
CURRENCYCODE has duplicate values --- Constraint --- 1

Validation script--


UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'CURRENCYCODE is blank','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Error', '; ')
	FROM PWC_E_CURRENCIES
	WHERE 
		CURRENCYCODE= '' or CURRENCYCODE is null

UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'REFERENCECURRENCYFORTRIANGULATION is blank','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Warning', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		REFERENCECURRENCYFORTRIANGULATION= '' or REFERENCECURRENCYFORTRIANGULATION is null

UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'GENERALROUNDINGRULE is blank','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Warning', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		GENERALROUNDINGRULE= '' or GENERALROUNDINGRULE is null

UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'SYMBOL is blank','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Warning', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		SYMBOL= '' or SYMBOL is null

UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'NAME is blank','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Warning', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		NAME= '' or NAME is null

UPDATE PWC_E_CURRENCIES 
	SET ERRORDESC = CONCAT(ERRORDESC,'CURRENCYCODE,DATAAREAID has duplicate values','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Constraint', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		CURRENCYCODE IS NOT NULL AND DATAAREAID IS NOT NULL AND CONCAT(CURRENCYCODE,DATAAREAID) IN 
		(SELECT CONCAT(CURRENCYCODE,DATAAREAID) FROM PWC_E_CURRENCIES
		GROUP BY CURRENCYCODE,DATAAREAID HAVING COUNT(*)>1)
