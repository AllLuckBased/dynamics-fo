Currencies ERP Length Pilot Validation

UPDATE PWC_E_CURRENCIES
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'CURRENCYCODE exceeds max length 3','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		LEN(CURRENCYCODE) > 3

UPDATE PWC_E_CURRENCIES
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'SYMBOL exceeds max length 5','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		LEN(SYMBOL) > 5

UPDATE PWC_E_CURRENCIES
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'NAME exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CURRENCIES	
	WHERE 
		LEN(NAME) > 60
