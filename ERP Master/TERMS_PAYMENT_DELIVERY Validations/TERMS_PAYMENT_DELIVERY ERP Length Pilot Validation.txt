TERMS_PAYMENT_DELIVERY ERP Length Pilot Validation


TERMS OF PAYMENT---

UPDATE PWC_E_TERMSOFPAYMENT
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'CashPaymentMainAccountIdDisplayValue exceeds max length 500','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_TERMSOFPAYMENT	
	WHERE 
		LEN(CashPaymentMainAccountIdDisplayValue) > 500

UPDATE PWC_E_TERMSOFPAYMENT
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Description exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_TERMSOFPAYMENT	
	WHERE 
		LEN(Description) > 60

UPDATE PWC_E_TERMSOFPAYMENT
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Name exceeds max length 100','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_TERMSOFPAYMENT	
	WHERE 
		LEN(Name) > 100



TERMS OF DELIVERY---

UPDATE PWC_E_TERMSOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TermsCode exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_TERMSOFDELIVERY	
	WHERE 
		LEN(TermsCode) > 10

UPDATE PWC_E_TERMSOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TermsDescription exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_TERMSOFDELIVERY	
	WHERE 
		LEN(TermsDescription) > 60
