Modes Of Delivery ERP Length Pilot Validation

UPDATE PWC_E_MODEOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'MODECODE exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_MODEOFDELIVERY	
	WHERE 
		LEN(MODECODE) > 10

UPDATE PWC_E_MODEOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'MODEDESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_MODEOFDELIVERY	
	WHERE 
		LEN(MODEDESCRIPTION) > 60
