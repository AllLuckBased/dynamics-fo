Modes Of Delivery ERP Special Character Check Pilot Validation

UPDATE PWC_E_MODEOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'MODECODE has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_MODEOFDELIVERY
	WHERE 
	MODECODE LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_MODEOFDELIVERY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'MODEDESCRIPTION has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_MODEOFDELIVERY
	WHERE 
	MODEDESCRIPTION LIKE '%[^A-Za-z0-9 -]%'
