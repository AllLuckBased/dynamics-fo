Financial Period Close Template Tasks ERP Length Pilot Validation



Financial period close template tasks----

UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Template exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	WHERE 
		LEN(Template) > 60


UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Task exceeds max length 255','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	WHERE 
		LEN(Task) > 255


UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK

	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK

		LEN(TaskLink) > 255

UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
