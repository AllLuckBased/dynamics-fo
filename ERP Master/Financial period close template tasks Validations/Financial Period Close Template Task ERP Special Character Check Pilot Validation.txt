Financial Period Close Template Tasks ERP Special Character Check Pilot Validation




Financial period close template tasks----

UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Template has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	WHERE 
	Template LIKE '%[^A-Za-z0-9 -]%'

UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'Task has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	WHERE 
	Task LIKE '%[^A-Za-z0-9 -]%'

UPDATE PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TaskLink has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_LEDGERPERIODCLOSETEMPLATETASK
	WHERE 
	TaskLink LIKE '%[^A-Za-z0-9 -]%'
