Production Route Length ERP Plot Validation



-------------------Route Groups

UPDATE PWC_E_ROUTEGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPID exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_ROUTEGROUPS	
	WHERE LEN(GROUPID) > 10 

UPDATE PWC_E_ROUTEGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPNAME exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_ROUTEGROUPS	
	WHERE LEN(GROUPNAME) > 60
