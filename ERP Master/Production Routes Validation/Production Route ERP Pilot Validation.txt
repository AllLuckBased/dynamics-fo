		(SELECT B.CalendarId FROM PWC_E_PRODCALENDAR B WHERE (B.DATAAREAID = A.DATAAREAID OR UPPER(B.DATAAREAID) = 'GLOBAL') AND B.CalendarId IS NOT NULL)
