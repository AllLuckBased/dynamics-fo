Production Route ERP Special Character Check Pilot Validation



Route Groups--


UPDATE PWC_E_ROUTEGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPID has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_ROUTEGROUPS
	WHERE 
	GROUPID LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_ROUTEGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPNAME has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_ROUTEGROUPS
	WHERE 
	GROUPNAME LIKE '%[^A-Za-z0-9 -]%'
