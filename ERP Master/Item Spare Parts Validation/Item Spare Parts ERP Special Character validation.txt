Item Spare Parts ERP Special Characters Validation


-------------- Released products V2

UPDATE PWC_E_ECORESRELEASEDPRODUCTV2STAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'ITEMNUMBER has special character and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_ECORESRELEASEDPRODUCTV2STAGING	
	WHERE ITEMNUMBER LIKE '%[^A-Za-z0-9-]%'
