Users ERP Pilot Length Validation:

----- User Information

UPDATE PWC_E_USERINFO
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'USERID exceeds max length 80','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_USERINFO	
	WHERE LEN(USERID) > 80
