Users ERP Pilot Validation:


----------- Validation Rules-----------


----- User Information

USERID is null --- Error --- 2
USERNAME is null ----- Error ---- 2
NETWORKDOMAIN is null ----- Error ---- 2
EMAIL is null --- Warning --- 3
COMPANY is null --- Warning --- 3
ALIAS is null --- Warning --- 3
USERINFO_LANGUAGE is null --- Warning --- 3

USERID has duplicate values --- Constraint --- 1

COMPANY is not there in LEGALENTITYID of legal entities --- Error --- 2
UserInfo_language is not there in LanguageCodeId of Language Codes --- Error --- 2




----- User Roles

SECURITYROLENAME is null --- Error --- 2
SECURITYROLEIDENTIFIER is null --- Warning --- 3
USERLICENSETYPE is null --- Warning --- 3
DESCRIPTION is null --- Warning --- 3

SECURITYROLENAME has duplcate values --- Constraint --- 1

UserLicenseType
USERLICENSETYPE has invalid enum values --- Error --- 2





----- Warehouse Workers

PWC_E_EMPLOYEE
WAREHOUSEWORKERPERSONNELNUMBER is null --- Error --- 2
WAREHOUSEWORKERPERSONNELNUMBER has duplicate values --- Constraint ---- 1
WAREHOUSEWORKERPERSONNELNUMBER is not there in PERSONNELNUMBER of Employee --- Error --- 2

------- Warehouse Worker

UPDATE PWC_E_WAREHOUSEWORKER
	SET ERRORDESC = CONCAT(ERRORDESC,'WAREHOUSEWORKERPERSONNELNUMBER is blank','; ')
	,ERROR_VALUE = WAREHOUSEWORKERPERSONNELNUMBER
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Error', '; ')
	FROM PWC_E_WAREHOUSEWORKER	
	WHERE 
		WAREHOUSEWORKERPERSONNELNUMBER= '' or WAREHOUSEWORKERPERSONNELNUMBER is null

UPDATE PWC_E_WAREHOUSEWORKER
	SET ERRORDESC = CONCAT(ERRORDESC,'DEFAULTPACKINGSTATIONWAREHOUSEID is blank','; ')
	,ERROR_VALUE = WAREHOUSEWORKERPERSONNELNUMBER
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Error', '; ')
	FROM PWC_E_WAREHOUSEWORKER	
	WHERE 
		DEFAULTPACKINGSTATIONWAREHOUSEID= '' or DEFAULTPACKINGSTATIONWAREHOUSEID is null

UPDATE PWC_E_WAREHOUSEWORKER 
	SET ERRORDESC = CONCAT(ERRORDESC,'WAREHOUSEWORKERPERSONNELNUMBER,DATAAREAID has duplicate values','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Constraint', '; ')
	FROM PWC_E_WAREHOUSEWORKER	
	WHERE 
		WAREHOUSEWORKERPERSONNELNUMBER IS NOT NULL AND
		DATAAREAID IS NOT NULL AND
		CONCAT(WAREHOUSEWORKERPERSONNELNUMBER,DATAAREAID) IN 
		(SELECT CONCAT(WAREHOUSEWORKERPERSONNELNUMBER,DATAAREAID) FROM PWC_E_WAREHOUSEWORKER
		GROUP BY WAREHOUSEWORKERPERSONNELNUMBER,DATAAREAID HAVING COUNT(*)>1)

UPDATE PWC_E_WAREHOUSEWORKER
	SET ERRORDESC = CONCAT(ERRORDESC,'WAREHOUSEWORKERPERSONNELNUMBER is not there in PERSONNELNUMBER of Employee','; ')
	,ERROR_VALUE = WAREHOUSEWORKERPERSONNELNUMBER
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Constraint', '; ')
	FROM PWC_E_WAREHOUSEWORKER A
	WHERE 
		WAREHOUSEWORKERPERSONNELNUMBER IS NOT NULL AND WAREHOUSEWORKERPERSONNELNUMBER NOT IN 
		(SELECT B.PERSONNELNUMBER FROM PWC_E_EMPLOYEE B WHERE (B.DATAAREAID = A.DATAAREAID OR UPPER(B.DATAAREAID) = 'GLOBAL' OR UPPER(A.DATAAREAID) = 'GLOBAL') AND B.PERSONNELNUMBER IS NOT NULL)


UPDATE PWC_E_WAREHOUSEWORKER
	SET ERRORDESC = CONCAT(ERRORDESC,'DEFAULTPACKINGSTATIONWAREHOUSEID is not there in WAREHOUSEID of Warehouses','; ')
	,ERRORSTATUS = CONCAT(ERRORSTATUS, 'Error', '; ')
	FROM PWC_E_WAREHOUSEWORKER A
	WHERE 
		DEFAULTPACKINGSTATIONWAREHOUSEID IS NOT NULL AND DEFAULTPACKINGSTATIONWAREHOUSEID NOT IN 
		(SELECT B.WAREHOUSEID FROM PWC_E_WAREHOUSEERP B WHERE (B.DATAAREAID = A.DATAAREAID OR UPPER(B.DATAAREAID) = 'GLOBAL' OR UPPER(A.DATAAREAID) = 'GLOBAL') AND B.WAREHOUSEID IS NOT NULL)
