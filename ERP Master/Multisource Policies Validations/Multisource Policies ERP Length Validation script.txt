Multisource Policies ERP Length Validation Script


Multisource Policy---


UPDATE PWC_E_MULTISRCPOL 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'POLICYNAME exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_MULTISRCPOL	
	WHERE 
		LEN(POLICYNAME) > 60

UPDATE PWC_E_MULTISRCPOL 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'POLICYDESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_MULTISRCPOL	
	WHERE 
		LEN(POLICYDESCRIPTION) > 60
