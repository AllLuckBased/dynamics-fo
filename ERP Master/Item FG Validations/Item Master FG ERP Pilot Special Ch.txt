	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPID has special character and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_ITEMGROUPS
	WHERE GROUPID LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_ITEMGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GROUPNAME has special character','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_ITEMGROUPS
	WHERE GROUPNAME LIKE '%[^A-Za-z0-9 -]%'
