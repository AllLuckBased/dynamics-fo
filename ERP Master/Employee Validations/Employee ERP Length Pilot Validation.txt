UPDATE PWC_E_EMPLOYEE
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'PERSONNELNUMBER exceeds max length 25','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_EMPLOYEE
	WHERE 
		LEN(PERSONNELNUMBER) > 25
