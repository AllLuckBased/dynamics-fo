Employee ERP Special Character Check Pilot Validation



Employee V2---

UPDATE PWC_E_EMPLOYEE
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'PERSONNELNUMBER has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_EMPLOYEE
	WHERE 
	PERSONNELNUMBER LIKE '%[^A-Za-z0-9-]%'
