UPDATE PWC_E_LEGALENTITY
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'ADDRESSDESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_LEGALENTITY
	WHERE 
		LEN(ADDRESSDESCRIPTION) > 60
