Main Accounts ERP Length Pilot Validation

Chart of Accounts--

UPDATE PWC_E_CHARTOFACC
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'CHARTOFACCOUNTS exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CHARTOFACC
	WHERE 
		LEN(CHARTOFACCOUNTS) > 60

UPDATE PWC_E_CHARTOFACC
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'DESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CHARTOFACC
	WHERE 
		LEN(DESCRIPTION) > 60
