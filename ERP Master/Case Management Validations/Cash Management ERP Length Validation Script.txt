Cash Management ERP Length Validation Script

Case Categories---

UPDATE PWC_E_CASECATEGORIES 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'CASECATEGORY exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CASECATEGORIES	
	WHERE 
		LEN(CASECATEGORY) > 10


UPDATE PWC_E_CASECATEGORIES 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'DESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CASECATEGORIES	
	WHERE 
		LEN(DESCRIPTION) > 60
