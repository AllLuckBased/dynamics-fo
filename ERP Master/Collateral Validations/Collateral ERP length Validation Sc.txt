Collateral ERP length Validation Script



Insurance and guarantees------

UPDATE PWC_E_CREDMANCUSTGUARANTEEINSURANCE 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'GUARANTEENUMBER exceeds max length 20','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CREDMANCUSTGUARANTEEINSURANCE	
	WHERE 
		LEN(GUARANTEENUMBER) > 20


Insurance and guarantee types----

UPDATE PWC_E_CREDMANGUARANTEEINSURANCETYPE 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'DESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CREDMANGUARANTEEINSURANCETYPE	
	WHERE 
		LEN(DESCRIPTION) > 60

UPDATE PWC_E_CREDMANGUARANTEEINSURANCETYPE 
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TYPE exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_CREDMANGUARANTEEINSURANCETYPE	
	WHERE 
		LEN(TYPE) > 10
