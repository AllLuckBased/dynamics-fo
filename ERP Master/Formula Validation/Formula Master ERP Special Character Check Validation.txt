Formula Master ERP Special Character Check Pilot Validation


Formula Header--

UPDATE PWC_E_PMFFORMULAHEADERSTAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'FORMULAID has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_PMFFORMULAHEADERSTAGING
	WHERE 
	FORMULAID LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_PMFFORMULAHEADERSTAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'FORMULANAME has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_PMFFORMULAHEADERSTAGING
	WHERE 
	FORMULANAME LIKE '%[^A-Za-z0-9 -]%'

UPDATE PWC_E_PMFFORMULAHEADERSTAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'OldBOMCode has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_PMFFORMULAHEADERSTAGING
	WHERE 
	OldBOMCode LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_PMFFORMULAHEADERSTAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'LegacyFormulaId has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_PMFFORMULAHEADERSTAGING
	WHERE 
	LegacyFormulaId LIKE '%[^A-Za-z0-9-]%'
