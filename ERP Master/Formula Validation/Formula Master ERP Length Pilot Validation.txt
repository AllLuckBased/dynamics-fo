Formula Master ERP Length Pilot Validation


Formula Header--
UPDATE PWC_E_PMFFORMULAHEADERSTAGING
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'FORMULAID exceeds max length 20','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_PMFFORMULAHEADERSTAGING
	WHERE 
		LEN(FORMULAID) > 20

		LEN(FORMULANAME) > 60
