Address ERP Length Pilot Validation


Country--
UPDATE PWC_E_COUNTRYREGION
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'COUNTRYREGION exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_COUNTRYREGION
	WHERE 
		LEN(COUNTRYREGION) > 10
