Address ERP Special Character Check Pilot Validation


Country--

UPDATE PWC_E_COUNTRYREGION
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'COUNTRYREGION has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_COUNTRYREGION
	WHERE 
	COUNTRYREGION LIKE '%[^A-Za-z0-9-]%'
