Sales Tax ERP Length Pilot Validation


Sales Tax Group--

UPDATE PWC_E_SALESTAXGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TAXGROUPCODE exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_SALESTAXGROUPS
	WHERE 
		LEN(TAXGROUPCODE) > 10

UPDATE PWC_E_SALESTAXGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'DESCRIPTION exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_SALESTAXGROUPS
	WHERE 
		LEN(DESCRIPTION) > 60
