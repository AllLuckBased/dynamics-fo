Sales Tax ERP Special Character Check Pilot Validation

Sales Tax Group--

UPDATE PWC_E_SALESTAXGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'TAXGROUPCODE has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_SALESTAXGROUPS
	WHERE 
	TAXGROUPCODE LIKE '%[^A-Za-z0-9-]%'

UPDATE PWC_E_SALESTAXGROUPS
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'DESCRIPTION has special characters','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_SALESTAXGROUPS
	WHERE 
	DESCRIPTION LIKE '%[^A-Za-z0-9 -]%'
