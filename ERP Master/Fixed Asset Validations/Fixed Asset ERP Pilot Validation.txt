	SET C.ERRORDESC = CONCAT(ERRORDESC, 'MOVEMENTDATE should be strictly greater than ACQUISITIONDATE of the Fixed Asset','; ')
	,C.ERRORSTATUS = CONCAT(ERRORSTATUS, 'ERROR', '; ')

    ON (C.FIXEDASSETNUMBER = F.FIXEDASSETNUMBER AND C.DATAAREAID = F.DATAAREAID)
	WHERE
	ACQUISITIONDATE IS NOT NULL AND
    MOVEMENTDATE IS NOT NULL AND
    ACQUISITIONDATE >= MOVEMENTDATE
