Warehouse--

UPDATE PWC_E_WAREHOUSEERP
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'WAREHOUSEID has special characters and/or blank spaces','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Error', '; ')
	FROM PWC_E_WAREHOUSEERP
	WHERE 
	WAREHOUSEID LIKE '%[^A-Za-z0-9-]%'

	WAREHOUSENAME LIKE '%[^A-Za-z)0-9( -]%'
