Site Warehouse Location  Length ERP Pilot Validation script





----> Sites

UPDATE PWC_E_SITEERP
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'SiteId exceeds max length 10','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_SITEERP
	WHERE 
		LEN(SiteId) > 10

UPDATE PWC_E_SITEERP
	SET ERRORDESC2 = CONCAT(ERRORDESC2,'SiteName exceeds max length 60','; ')
	,ERRORSTATUS2 = CONCAT(ERRORSTATUS2, 'Constraint', '; ')
	FROM PWC_E_SITEERP
	WHERE 
		LEN(SiteName) > 60
